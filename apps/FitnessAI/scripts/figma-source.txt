  className?: string; outline?: boolean; sm?: boolean;
}) {

function ProgressRing({ progress, size = 120, sw = 10, color = "#0066FF", bg = "rgba(255,255,255,0.07)", children }: {
  progress: number; size?: number; sw?: number; color?: string; bg?: string; children?: React.ReactNode;
}) {
  const r = (size - sw) / 2;
  const circ = 2 * Math.PI * r;
  const dash = circ * Math.min(progress, 100) / 100;
  return (
    <div className="relative flex-shrink-0" style={{ width: size, height: size }}>
      <svg width={size} height={size} style={{ transform: "rotate(-90deg)", position: "absolute" }}>
        <circle cx={size / 2} cy={size / 2} r={r} fill="none" stroke={bg} strokeWidth={sw} />
        <circle cx={size / 2} cy={size / 2} r={r} fill="none" stroke={color} strokeWidth={sw}
          strokeDasharray={`${dash} ${circ}`} strokeLinecap="round"
          style={{ filter: `drop-shadow(0 0 6px ${color}90)`, transition: "stroke-dasharray 0.6s ease" }}
        />
      </svg>
      <div className="absolute inset-0 flex items-center justify-center">{children}</div>
    </div>
  );
}

// ─── Bottom Navigation ────────────────────────────────────────────────────────
function BottomNav({ screen, setScreen, lang }: { screen: AppScreen; setScreen: (s: AppScreen) => void; lang: Lang }) {
  const t = T[lang];
  const items: { id: AppScreen; icon: React.ReactNode; label: string }[] = [
    { id: "dashboard", icon: <Home size={22} />, label: t.home },
    { id: "nutrition", icon: <Utensils size={22} />, label: t.nutrition },
    { id: "workout", icon: <Dumbbell size={22} />, label: t.workout },
    { id: "ai", icon: <Sparkles size={22} />, label: t.ai },
    { id: "profile", icon: <User size={22} />, label: t.profile },
  ];
  return (
    <div className="fixed bottom-0 left-1/2 -translate-x-1/2 w-full max-w-[430px] z-50 px-4 pb-6 pt-2"
      style={{ background: "linear-gradient(to top, #050A14 60%, transparent)" }}>
      <div className="flex justify-around items-center px-2 py-3 rounded-3xl" style={{ background: "rgba(10,22,40,0.95)", backdropFilter: "blur(24px)", border: "1px solid rgba(255,255,255,0.08)", boxShadow: "0 -4px 40px rgba(0,0,0,0.5)" }}>
        {items.map(item => {
          const active = screen === item.id;
          return (
            <button key={item.id} onClick={() => setScreen(item.id)}
              className="flex flex-col items-center gap-1 px-3 py-1.5 rounded-2xl transition-all active:scale-90"
              style={active ? { background: GRAD_SOFT } : {}}>
              <div style={{ color: active ? "#06B6D4" : "rgba(255,255,255,0.35)", filter: active ? "drop-shadow(0 0 8px #06B6D450)" : "none" }}>
                {item.icon}
              </div>
              <span className="text-xs font-semibold" style={{ color: active ? "#06B6D4" : "rgba(255,255,255,0.35)", fontFamily: "var(--font-display)" }}>
                {item.label}
              </span>
            </button>
          );
        })}
      </div>
    </div>
  );
}

// ─── Screen Header (for main app screens) ────────────────────────────────────
function AppHeader({ screen, lang, setScreen }: { screen: AppScreen; lang: Lang; setScreen: (s: AppScreen) => void }) {
  const secondaryScreens: Partial<Record<AppScreen, boolean>> = { progress: true, analytics: true, health: true, premium: true };
  if (["dashboard", "nutrition", "workout", "ai", "profile"].includes(screen)) return null;
  return (
    <div className="px-5 pt-2 pb-2 flex items-center gap-3" style={{ borderBottom: "1px solid rgba(255,255,255,0.05)" }}>
      <div className="flex gap-2 overflow-x-auto">
        {(["progress", "analytics", "health", "premium"] as AppScreen[]).map(s => (
          <button key={s} onClick={() => setScreen(s)}
            className="flex-shrink-0 px-4 py-2 rounded-xl text-xs font-bold transition-all"
            style={screen === s ? { background: GRAD, color: "white", boxShadow: "0 2px 12px rgba(0,102,255,0.3)" } : { ...GLASS, color: "rgba(255,255,255,0.4)" }}>
            {T[lang][s] || s}
          </button>
        ))}
      </div>
    </div>
  );
}

        {flow === "auth" && <AuthScreen lang={lang} onDone={() => setFlow("app")} />}

        {flow === "app" && (

                    style={{ borderBottom: "1px solid rgba(255,255,255,0.06)" }}>
                    <div className="flex items-center gap-2">
                      {/* Back to Home button */}

                      {(["progress", "analytics", "health", "premium"] as AppScreen[]).includes(appScreen) && (
                        <div className="flex gap-1.5 overflow-x-auto ml-1">
                          {(["progress", "analytics", "health", "premium"] as AppScreen[]).map(s => (
                            <button key={s} onClick={() => handleSetScreen(s)}
                              className="flex-shrink-0 px-3 py-2 rounded-xl text-xs font-bold transition-all"
